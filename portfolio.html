Niveus AI

An AI-powered Discord bot that integrates large language model capabilities directly into Discord servers. Built to handle natural language queries, context-aware conversations, and automated responses — all within the Discord interface.

Python Discord.py OpenAI API AsyncIO
import discord
import openai
import asyncio
import os
from dotenv import load_dotenv
load_dotenv()

openai.api_key = os.getenv("OPENAI_API_KEY")
# Context-aware conversation handler
conversation_history = {}

async def ask_ai(user_id, prompt):
    history = conversation_history.get(user_id, [])
    history.append({"role": "user", "content": prompt})
    response = await openai.ChatCompletion.acreate(
        model="gpt-4", messages=history
    )
    reply = response.choices[0].message.content
    history.append({"role": "assistant", "content": reply})
    conversation_history[user_id] = history
    return reply
